  - pace_value: string | number
  - financial_value: string | number
  - prospectus_value: string | number

  - status: 'aligned' | 'needs_review' | 'critical'
  - variance_percent: number (0-100)
  - isExplained: boolean

  - lastUpdated?: Date

  - severity: 'critical' | 'warning' | 'info'
  - metric: string

  - value1: string | number

  - impact: string
  - suggestedFix: string

AlertRule
  - metric: string
  - max_variance_percent: number
  - enabled: boolean

================================================================================
SEVERITY CLASSIFICATION
================================================================================

🟢 ALIGNED (Green)
  - Variance: < 5%
  - Status: bg-green-50, text-green-700
  - Icon: CheckCircle2
  - Meaning: Data sources are internally consistent

🟡 NEEDS REVIEW (Yellow)
  - Variance: 5-10%
  - Status: bg-yellow-50, text-yellow-700
  - Icon: AlertTriangle
  - Meaning: Variance exists, needs documentation

🔴 CRITICAL (Red)
  - Variance: > 10%
  - Status: bg-red-50, text-red-700
  - Icon: AlertCircle
  - Meaning: Significant conflict requiring immediate attention
